元素的添加和删除；

  1. 元素的添加； 
                        var arr = ['h', 'q', 'y', 'j'];
                    arr[4] = 'five'; //指定索引，添加元素
                    console.log(arr);

                    arr.push('one');//在末尾添加元素
                    console.log(arr);
                    arr.push('two_1', 'two_2');
                    console.log(arr);

                    arr.unshift('first');//在开头添加元素
                    console.log(arr);
                    arr.unshift('first_1', 'first_2');//两个元素一次加入
                    console.log(arr);
  2. 元素的删除： 
                        var arr = ['h', 'q', 'y', 'j'];
                    delete arr[2]; //删除一个指定元素，注意和下面的区别
                    console.log(arr);
                    arr.pop();//删除最后一个元素
                    console.log(arr); 

                    arr.shift();//删除第一个元素
                    console.log(arr); 

                    delete arr[0]; //删除一个指定元素
                    console.log(arr);
