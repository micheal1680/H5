    复杂点的高阶函数； 
                        //检查一个值是否为偶数
                    function even (a) {
                        return a%2 === 0;
                    }
                    //对函数判断进行反转
                    function not (f) {
                        return function () {
                            var result = f.apply(this, arguments);
                            console.log(!result);//这里会打印什么那？我很好奇
                            return !result;
                        }
                    }

                    var odd     = not(even);
                    console.log(odd);
                    var arr     = [1,3,5,4];
                    var r1      = arr.every(odd);
                    var r2      = arr.some(odd);
                    console.log(r1);//false
                    console.log(r2);//true
