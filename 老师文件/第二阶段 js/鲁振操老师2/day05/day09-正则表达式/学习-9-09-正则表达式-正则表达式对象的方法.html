正则表达式的方法；

  1. exec(str)：参数是字符串，对该字符串进行匹配检索；
    总是返回一个结果，并提供匹配的完整信息；再次执行会进行下次匹配；
    没找到返回null； 
                        var pattern     = /hqyj|farsight/ig; //如果去掉i 或 j 会怎样？
                    var str         = '华清远见的简称是HQyj还是hqyj那还是farsight那？';
                    var allmatch1   = pattern.exec(str);
                    var allmatch2   = pattern.exec(str);
                    var allmatch3   = pattern.exec(str);
                    var allmatch4   = pattern.exec(str);
                    console.log(allmatch1);
                    console.log(allmatch2);
                    console.log(allmatch3);
                    console.log(allmatch4);
  2. test(str)：对字符串进行检测，包含返回true，否则返回false； 
                        var pattern     = /hqyj|farsight/ig; //如果去掉i 或 j 会怎样？
                    var str         = '华清远见的简称是HQyj还是hqyj那还是farsight那？';
                    var allmatch1   = pattern.test(str);
                    var allmatch2   = pattern.test(str);
                    var allmatch3   = pattern.test(str);
                    var allmatch4   = pattern.test(str);
                    console.log(allmatch1);
                    console.log(allmatch2);
                    console.log(allmatch3);
                    console.log(allmatch4);
